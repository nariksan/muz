async function loadAllCommentCounts() {
    if (!db) return;
    try {
        const snapshot = await db.collection('comments').get();
        const counts = {};
        snapshot.forEach(doc => {
            const data = doc.data();
            const key = String(data.trackId);
            counts[key] = (counts[key] || 0) + 1;
        });
        Object.assign(commentCountsCache, counts);
        updateAllCommentBadges();
        renderNewRelease();
    } catch (e) {
        console.error('Ошибка загрузки счётчиков:', e);
    }
}

function updateAllCommentBadges() {
    document.querySelectorAll('.track-comment-btn, .special-card-btn[data-action="comment"]').forEach(btn => {
        const trackId = String(btn.dataset.trackId);
        const count = commentCountsCache[trackId] || 0;
        const badge = btn.querySelector('.comment-count-badge');
        if (badge) badge.textContent = count > 0 ? count : '';
    });
}

function openCommentsModal(track, trackType) {
    currentCommentTrack = { ...track, trackType };
    const modal = document.getElementById('commentsModal');
    const list = document.getElementById('commentsList');
    const info = document.getElementById('commentsTrackInfo');
    const countEl = document.getElementById('commentsCount');

    const typeLabels = { tracks: 'Трек', instrumentals: 'Инструментал', subscribers: 'Дай голос' };
    const icon = trackType === 'instrumentals' ? '🎹' : trackType === 'subscribers' ? '' : '🎵';
    const coverHtml = track.cover ? `<img src="${track.cover}" onerror="this.style.display='none';this.parentElement.innerHTML='${icon}'">` : icon;
    const artistText = trackType === 'subscribers' ? track.author : 'EVGENII';

    info.innerHTML = `<div class="comments-track-cover">${coverHtml}</div><div><div class="comments-track-name">${track.title}</div><div class="comments-track-type">${typeLabels[trackType]} · ${artistText}</div></div>`;

    list.innerHTML = '<div class="comments-loading">Загрузка комментариев</div>';
    countEl.textContent = '';
    modal.classList.add('active');
    document.body.style.overflow = 'hidden';

    if (commentsUnsubscribe) commentsUnsubscribe();
    
    commentsUnsubscribe = db.collection('comments')
        .where('trackId', '==', String(track.id))
        .onSnapshot(snapshot => {
            const comments = [];
            snapshot.forEach(doc => comments.push({ id: doc.id, ...doc.data() }));
            
            comments.sort((a, b) => {
                const dateA = a.createdAt?.toDate ? a.createdAt.toDate() : new Date(a.createdAt || 0);
                const dateB = b.createdAt?.toDate ? b.createdAt.toDate() : new Date(b.createdAt || 0);
                return dateA - dateB;
            });
            
            countEl.textContent = comments.length > 0 ? `(${comments.length})` : '';
            renderComments(comments);
        }, error => {
            console.error('Ошибка подписки:', error);
            list.innerHTML = '<div class="comments-empty"><div class="comments-empty-icon">⚠️</div>Ошибка: ' + error.message + '</div>';
        });

    subscribeToUserLikes();
    setTimeout(() => document.getElementById('commentInput').focus(), 300);
}

    if (comments.length === 0) {

        const time = formatCommentTime(c.createdAt);

        const isLiked = !!userLikesCache[c.id];
        const likeClass = isLiked ? ' liked' : '';

function formatCommentTime(timestamp) {
    if (!timestamp) return '';
    const date = timestamp.toDate ? timestamp.toDate() : new Date(timestamp);
    const now = new Date();
    const diff = Math.floor((now - date) / 1000);
    if (diff < 60) return 'только что';
    if (diff < 3600) return `${Math.floor(diff / 60)} мин назад`;
    if (diff < 86400) return `${Math.floor(diff / 3600)} ч назад`;
    if (diff < 604800) return `${Math.floor(diff / 86400)} дн назад`;
    return date.toLocaleDateString('ru-RU', { day: 'numeric', month: 'short' });
}

async function sendComment() {
    if (!db || !currentCommentTrack) return;
    const input = document.getElementById('commentInput');
    const text = input.value.trim();
    if (!text) return;

    const userName = localStorage.getItem('evgenii_user') || 'Гость';
    const btn = document.getElementById('commentSendBtn');
    btn.disabled = true;

    try {
        await db.collection('comments').add({
            trackId: String(currentCommentTrack.id),
            trackType: currentCommentTrack.trackType,
            trackTitle: currentCommentTrack.title,
            userName: userName,
            text: text,
            likesCount: 0,
            createdAt: firebase.firestore.FieldValue.serverTimestamp()
        });
        input.value = '';
        input.style.height = 'auto';
        commentCountsCache[String(currentCommentTrack.id)] = (commentCountsCache[String(currentCommentTrack.id)] || 0) + 1;
        updateAllCommentBadges();
        renderNewRelease();
    } catch (e) {
        console.error('Ошибка отправки:', e);
        alert('Ошибка отправки комментария: ' + e.message);
    } finally {
        btn.disabled = false;
    }
}

let userLikesCache = {};
let likesUnsubscribe = null;

function subscribeToUserLikes() {
    if (likesUnsubscribe) likesUnsubscribe();
    
    likesUnsubscribe = db.collection('likes')
        .where('userId', '==', CURRENT_USER_ID)
        .onSnapshot(snapshot => {
            userLikesCache = {};
            snapshot.forEach(doc => {
                const data = doc.data();
                userLikesCache[data.commentId] = doc.id;
            });
            updateLikeButtons();
        }, error => {
            console.error('Ошибка подписки на лайки:', error);
        });
}

async function toggleLike(commentId, commentDocId) {
    if (!db) return;
    const btn = document.querySelector(`.like-btn[data-comment-id="${commentId}"]`);
    if (btn) btn.disabled = true;
    
    try {
        const likesRef = db.collection('likes');
        const commentRef = db.collection('comments').doc(commentDocId);
        
        const existingLike = userLikesCache[commentId];
        
        if (existingLike) {
            await db.runTransaction(async (transaction) => {
                const commentDoc = await transaction.get(commentRef);
                if (!commentDoc.exists) throw new Error('Комментарий не найден');
                
                const currentCount = commentDoc.data().likesCount || 0;
                const newCount = Math.max(0, currentCount - 1);
                
                transaction.delete(likesRef.doc(existingLike));
                transaction.update(commentRef, { likesCount: newCount });
            });
            
            delete userLikesCache[commentId];
        } else {
            await db.runTransaction(async (transaction) => {
                const commentDoc = await transaction.get(commentRef);
                if (!commentDoc.exists) throw new Error('Комментарий не найден');
                
                const currentCount = commentDoc.data().likesCount || 0;
                
                const newLikeRef = likesRef.doc();
                transaction.set(newLikeRef, {
                    commentId: commentId,
                    userId: CURRENT_USER_ID,
                    createdAt: firebase.firestore.FieldValue.serverTimestamp()
                });
                transaction.update(commentRef, { likesCount: currentCount + 1 });
            });
            
            const snapshot = await likesRef
                .where('commentId', '==', commentId)
                .where('userId', '==', CURRENT_USER_ID)
                .limit(1)
                .get();
            snapshot.forEach(doc => {
                userLikesCache[commentId] = doc.id;
            });
        }
        
        updateLikeButtons();
    } catch (e) {
        console.error('Ошибка лайка:', e);
        alert('Ошибка: ' + e.message);
    } finally {
        if (btn) btn.disabled = false;
    }
}

function updateLikeButtons() {
    document.querySelectorAll('.like-btn').forEach(btn => {
        const commentId = btn.dataset.commentId;
        const isLiked = !!userLikesCache[commentId];
        btn.classList.toggle('liked', isLiked);
        btn.innerHTML = isLiked 
            ? '<svg viewBox="0 0 24 24"><path d="M12 21.35l-1.45-1.32C5.4 15.36 2 12.28 2 8.5 2 5.42 4.42 3 7.5 3c1.74 0 3.41.81 4.5 2.09C13.09 3.81 14.76 3 16.5 3 19.58 3 22 5.42 22 8.5c0 3.78-3.4 6.86-8.55 11.54L12 21.35z"/></svg>'
            : '<svg viewBox="0 0 24 24"><path d="M16.5 3c-1.74 0-3.41.81-4.5 2.09C10.91 3.81 9.24 3 7.5 3 4.42 3 2 5.42 2 8.5c0 3.78 3.4 6.86 8.55 11.54L12 21.35l1.45-1.32C18.6 15.36 22 12.28 22 8.5 22 5.42 19.58 3 16.5 3zm-4.4 15.55l-.1.1-.1-.1C7.14 14.24 4 11.39 4 8.5 4 6.5 5.5 5 7.5 5c1.54 0 3.04.99 3.57 2.36h1.87C13.46 5.99 14.96 5 16.5 5c2 0 3.5 1.5 3.5 3.5 0 2.89-3.14 5.74-7.9 10.05z"/></svg>';
        
        const countSpan = btn.querySelector('.like-count');
        if (countSpan) {
            const commentItem = btn.closest('.comment-item');
            if (commentItem) {
                const allBtns = commentItem.querySelectorAll('.like-btn');
                allBtns.forEach(b => {
                    const cs = b.querySelector('.like-count');
                    if (cs) {
                        const currentText = cs.textContent;
                        const currentNum = parseInt(currentText) || 0;
                        cs.textContent = currentNum > 0 ? currentNum : '';
                    }
                });
            }
        }
    });
}

    const badgeClass = category === 'instrumentals' ? 'gold' : category === 'subscribers' ? 'pink' : 'cyan';

    const commentCount = commentCountsCache[String(track.id)] || 0;
    el.innerHTML = `<div class="track-cover" onclick="${playFn}">${coverHtml}<div class="track-category-badge ${badgeClass}">${badgeIcon}</div></div><div class="track-info" onclick="${playFn}"><div class="track-title">${track.title}</div><div class="track-artist ${artistClass}">${artistText}</div></div><div class="track-duration">${track.duration}</div><button class="track-comment-btn" data-track-id="${track.id}" onclick="event.stopPropagation();openCommentsModal(${dataArray}[${index}],'${category}')"><svg viewBox="0 0 24 24"><path d="M21 15a2 2 0 0 1-2 2H7l-4 4V5a2 2 0 0 1 2-2h14a2 2 0 0 1 2 2z"/></svg><span class="comment-count-badge">${commentCount > 0 ? commentCount : ''}</span></button><button class="track-download-btn" onclick="event.stopPropagation();openDownloadModal(${dataArray}[${index}],'${downloadType}')"><svg viewBox="0 0 24 24"><path d="M19 9h-4V3H9v6H5l7 7 7-7zM5 18v2h14v-2H5z"/></svg></button>`;

    }
}

let currentPage = 'homePage', pageHistory = ['homePage'];
function navigateTo(pageId) {
    if (currentPage === pageId) return;
    const ce = document.getElementById(currentPage), ne = document.getElementById(pageId);
    pageHistory.push(pageId);
    ce.classList.remove('active');
    setTimeout(() => ce.classList.add('slide-out-left'), 50);
    ne.classList.remove('slide-out-left', 'slide-in-right');
    ne.classList.add('slide-in-right');
    setTimeout(() => { ne.classList.add('active'); ne.classList.remove('slide-in-right'); ne.scrollTop = 0; }, 50);
    currentPage = pageId;
}
function goBack() {
    if (pageHistory.length > 1) {
        pageHistory.pop();
        const pp = pageHistory[pageHistory.length - 1];
        const ce = document.getElementById(currentPage), pe = document.getElementById(pp);
        ce.classList.remove('active'); ce.classList.add('slide-in-right');
        setTimeout(() => { ce.classList.remove('slide-in-right'); pe.classList.remove('slide-out-left'); pe.classList.add('active'); }, 50);
        currentPage = pp;
    }
}

        } catch (error) { console.error('Ошибка отправки:', error); voiceFormStatus.textContent = `❌ Ошибка: ${error.message}`; voiceFormStatus.className = 'form-status error'; }
        finally { voiceSubmitBtn.disabled = false; voiceSubmitBtn.innerHTML = '<span>ОТПРАВИТЬ ТЕКСТ</span>'; }
    });
}
